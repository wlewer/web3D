## 2. IComponentPlugin 接口完整定义

```typescript
// src/types/builder.ts

export interface IComponentPlugin {
  /** 全局唯一标识 */
  id: string;
  /** 显示名称 */
  name: string;
  /** 组件分类 */
  category: ComponentCategory;
  /** 组件图标 (antd icon name) */
  icon: string;
  /** 组件描述 */
  description?: string;
  /** 默认属性 */
  defaultProps: Record<string, unknown>;
  /** 属性配置 Schema */
  propSchema: PropSchema[];
  /** 渲染组件 */
  component: React.ComponentType<any>;
  /** 是否是容器组件 */
  isContainer?: boolean;
  /** 允许嵌套的子组件类型 */
  allowedChildren?: string[];
  /** 是否支持响应式 */
  responsive?: boolean;
  /** 版本号 */
  version: string;
  /** 来源：内置 / 插件市场 */
  source: 'builtin' | 'marketplace';
  /** 插件包 ID（marketplace 来源时必填） */
  packageId?: string;
}

export type ComponentCategory =
  | 'basic'      // 基础
  | 'layout'     // 布局
  | '3d'         // 3D 模型
  | 'form'       // 表单
  | 'chart'      // 图表
  | 'media'      // 媒体
  | 'custom';    // 自定义

export interface PropSchema {
  /** 属性 key */
  key: string;
  /** 显示标签 */
  label: string;
  /** 编辑器类型 */
  type: PropEditorType;
  /** 默认值 */
  defaultValue?: unknown;
  /** 是否必填 */
  required?: boolean;
  /** 枚举选项 */
  options?: { label: string; value: unknown }[];
  /** 条件显示 */
  visibleWhen?: { key: string; equals: unknown };
  /** 校验规则 */
  rules?: ValidationRule[];
  /** 分组 */
  group?: 'general' | 'style' | 'data' | 'event';
}

export type PropEditorType =
  | 'text' | 'textarea' | 'number' | 'boolean'
  | 'select' | 'color' | 'image' | 'json'
  | 'model-picker' | 'slider' | 'margin'
  | 'event-handler';

export interface ValidationRule {
  type: 'required' | 'min' | 'max' | 'regex' | 'custom';
  message: string;
  value?: unknown;
}

export interface IComponentNode {
  id: string;
  pluginId: string;
  props: Record<string, unknown>;
  children?: IComponentNode[];
  style?: React.CSSProperties;
  layout?: {
    x: number;
    y: number;
    w: number;
    h: number;
    static?: boolean;
  };
}
```

## 4. 画布引擎技术方案

### 4.1 技术选型

| 库 | 用途 | 版本 |
|---|---|---|
| react-dnd | 拖拽系统（组件面板 → 画布） | ^16.0 |
| react-grid-layout | 网格布局管理 | ^1.4 |
| @dnd-kit/core | 精细化拖拽（画布内排序） | ^6.1 |

### 4.2 拖拽流程

```
[组件面板]                    [画布 Grid]
     │                            │
     │  dragstart (pluginId)      │
     │ ──────────────────────────>│
     │                            │
     │         hover              │
     │ <──────────────────────────│
     │                            │
     │          drop              │
     │ ──────────────────────────>│
     │                            │
     │    生成 IComponentNode     │
     │    插入 store.components   │
     │                            │
     │    dispatch AddCommand     │
     │                            ▼
```

### 4.3 画布核心实现

```tsx
// src/features/builder-canvas/BuilderCanvas.tsx

import { useDrop } from 'react-dnd';
import GridLayout from 'react-grid-layout';
import { useBuilderStore } from '@/stores/builderStore';

const BuilderCanvas: React.FC = () => {
  const { components, selectedId, setSelectedId, addComponent } = useBuilderStore();

  const [, dropRef] = useDrop(() => ({
    accept: 'COMPONENT_PLUGIN',
    drop: (item: IComponentPlugin, monitor) => {
      const offset = monitor.getClientOffset();
      if (offset) {
        addComponent(item, offset);
      }
    },
  }));

  const handleLayoutChange = (layout: Layout[]) => {
    // 同步 layout 到组件树
  };

  return (
    <div ref={dropRef} className="builder-canvas">
      <GridLayout
        className="layout"
        layout={components.map(c => c.layout!)}
        cols={12}
        rowHeight={60}
        width={1920}
        onLayoutChange={handleLayoutChange}
      >
        {components.map(node => (
          <CanvasNode
            key={node.id}
            node={node}
            isSelected={node.id === selectedId}
            onClick={() => setSelectedId(node.id)}
          />
        ))}
      </GridLayout>
    </div>
  );
};
```

## 6. 属性编辑器设计（动态表单生成）

```tsx
// src/features/prop-editor/PropEditor.tsx

const PropEditor: React.FC<{ node: IComponentNode }> = ({ node }) => {
  const plugin = getPluginById(node.pluginId);
  const { updateComponentProps } = useBuilderStore();

  const groupedProps = groupBy(plugin.propSchema, 'group');

  return (
    <div className="prop-editor">
      {Object.entries(groupedProps).map(([group, schemas]) => (
        <CollapsePanel key={group} header={groupLabels[group]}>
          {schemas.map(schema => (
            <PropField
              key={schema.key}
              schema={schema}
              value={node.props[schema.key]}
              onChange={(val) => updateComponentProps(node.id, schema.key, val)}
            />
          ))}
        </CollapsePanel>
      ))}
    </div>
  );
};

// PropField 根据 type 动态渲染不同编辑器
const PropField: React.FC<PropFieldProps> = ({ schema, value, onChange }) => {
  switch (schema.type) {
    case 'text': return <Input value={value} onChange={onChange} />;
    case 'color': return <ColorPicker value={value} onChange={onChange} />;
    case 'model-picker': return <ModelPicker value={value} onChange={onChange} />;
    case 'event-handler': return <EventBinder value={value} onChange={onChange} />;
    // ... 其他类型
  }
};
```

## 7. 命令模式实现方案（撤销/重做）

```typescript
// src/stores/builderStore.ts

interface Command {
  type: 'ADD' | 'DELETE' | 'UPDATE_PROPS' | 'UPDATE_LAYOUT' | 'MOVE';
  execute: () => void;
  undo: () => void;
  timestamp: number;
}

interface BuilderState {
  components: IComponentNode[];
  history: Command[];
  historyIndex: number;
  // ...
}

const useBuilderStore = create<BuilderState & BuilderActions>((set, get) => ({
  components: [],
  history: [],
  historyIndex: -1,

  executeCommand: (cmd: Command) => {
    cmd.execute();
    set(state => ({
      history: [...state.history.slice(0, state.historyIndex + 1), cmd],
      historyIndex: state.historyIndex + 1,
    }));
  },

  undo: () => {
    const { history, historyIndex } = get();
    if (historyIndex < 0) return;
    history[historyIndex].undo();
    set({ historyIndex: historyIndex - 1 });
  },

  redo: () => {
    const { history, historyIndex } = get();
    if (historyIndex >= history.length - 1) return;
    history[historyIndex + 1].execute();
    set({ historyIndex: historyIndex + 1 });
  },
}));
```

## 10. 后端页面 API 设计

| 方法 | 路径 | 说明 |
|------|------|------|
| POST | `/api/v1/pages` | 创建页面 |
| GET | `/api/v1/pages` | 页面列表（带租户过滤） |
| GET | `/api/v1/pages/{id}` | 获取页面 Schema |
| PUT | `/api/v1/pages/{id}` | 更新页面 Schema |
| DELETE | `/api/v1/pages/{id}` | 删除页面 |
| POST | `/api/v1/pages/{id}/publish` | 发布页面 |
| POST | `/api/v1/pages/{id}/clone` | 复制页面 |
| GET | `/api/v1/pages/{id}/render` | 渲染页面（公开访问） |

**页面模型（SQLAlchemy）：**

```python
# backend/app/models/page.py

class Page(Base):
    __tablename__ = "pages"

    id = Column(Integer, primary_key=True)
    tenant_id = Column(Integer, ForeignKey("tenants.id"), nullable=False)
    title = Column(String(128), nullable=False)
    slug = Column(String(128), nullable=False)
    schema_json = Column(JSON, nullable=False)
    status = Column(String(20), default="draft")  # draft, published, archived
    published_at = Column(DateTime, nullable=True)
    version = Column(Integer, default=1)
    created_by = Column(Integer, ForeignKey("users.id"))
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)
```
