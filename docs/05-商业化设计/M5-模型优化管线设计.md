## 2. LOD 生成算法选型

| 级别 | 算法 | 目标顶点数 | 适用场景 |
|------|------|-----------|----------|
| LOD0 (原始) | - | 100% | 高质量预览 |
| LOD1 | Quadric Error Metrics | 50% | 标准查看 |
| LOD2 | Simplification + Remesh | 25% | 缩略图/列表 |
| LOD3 | Voxel-based | 10% | 快速预览 |

**实现方案：**

```python
# backend/app/services/lod_service.py

import subprocess
from pathlib import Path

class LODService:
    def generate_lods(self, input_path: Path, output_dir: Path) -> dict:
        """使用 gltf-transform 生成多级别 LOD"""
        results = {}

        # LOD0: 原始文件（Draco 压缩）
        lod0 = output_dir / "model_lod0.glb"
        subprocess.run([
            "gltf-transform", "draco", str(input_path), str(lod0),
            "--position-quantization-bits", "14"
        ])
        results['lod0'] = lod0

        # LOD1: 50% 简化
        lod1 = output_dir / "model_lod1.glb"
        subprocess.run([
            "gltf-transform", "simplify", str(input_path), str(lod1),
            "--ratio", "0.5",
            "--error", "0.001"
        ])
        subprocess.run([
            "gltf-transform", "draco", str(lod1), str(lod1)
        ])
        results['lod1'] = lod1

        # LOD2: 25% 简化
        lod2 = output_dir / "model_lod2.glb"
        subprocess.run([
            "gltf-transform", "simplify", str(input_path), str(lod2),
            "--ratio", "0.25",
            "--error", "0.005"
        ])
        subprocess.run([
            "gltf-transform", "draco", str(lod2), str(lod2)
        ])
        results['lod2'] = lod2

        return results
```

## 3. 纹理压缩技术方案

### 3.1 KTX2 + Basis Universal

```python
# backend/app/services/texture_service.py

class TextureService:
    def compress_textures(self, input_dir: Path, output_dir: Path):
        """将 PNG/JPEG 纹理压缩为 KTX2/Basis"""
        for tex_file in input_dir.glob("*.*"):
            if tex_file.suffix.lower() in ['.png', '.jpg', '.jpeg']:
                output = output_dir / f"{tex_file.stem}.ktx2"
                subprocess.run([
                    "basisu", str(tex_file),
                    "-ktx2",
                    "-uastc",
                    "-uastc_level", "2",
                    "-output_file", str(output)
                ])

    def apply_to_gltf(self, gltf_path: Path, texture_mapping: dict):
        """使用 gltf-transform 替换纹理引用"""
        subprocess.run([
            "gltf-transform", "texture-compress", str(gltf_path), str(gltf_path),
            "--target", "ktx2",
            "--quality", "auto"
        ])
```

### 3.2 压缩效果对比

| 格式 | 大小 | GPU 解压 | 兼容性 |
|------|------|----------|--------|
| PNG (原始) | 100% | CPU 解码 |  universal |
| JPEG | 30% | CPU 解码 | universal |
| KTX2/BasisU ETC1S | 15% | GPU 硬件 | WebGL 2.0+ |
| KTX2/BasisU UASTC | 25% | GPU 硬件 | WebGL 2.0+ |

## 4. Celery 任务编排

```python
# backend/app/services/pipeline_service.py
from celery import chain, group

class PipelineService:
    def create_pipeline(self, model_id: int, file_path: str):
        """创建模型优化管线任务链"""
        task_chain = chain(
            validate_file.s(model_id, file_path),
            extract_metadata.s(),
            group(
                generate_lods.s(),
                compress_textures.s(),
                convert_formats.s()
            ),
            validate_outputs.s(),
            upload_to_cdn.s(),
            update_model_status.s()
        )
        result = task_chain.apply_async()
        return result.id

# tasks.py
from celery import shared_task

@shared_task(bind=True, max_retries=3)
def validate_file(self, model_id: int, file_path: str):
    """验证文件完整性"""
    try:
        validator = FileValidator(file_path)
        validator.check_format()
        validator.scan_virus()
        return {'model_id': model_id, 'file_path': file_path, 'valid': True}
    except Exception as exc:
        raise self.retry(exc=exc, countdown=60)

@shared_task
def extract_metadata(data: dict):
    """提取元数据"""
    extractor = MetadataExtractor(data['file_path'])
    metadata = extractor.extract()
    return {**data, 'metadata': metadata}

@shared_task
def generate_lods(data: dict):
    """生成 LOD"""
    lod_service = LODService()
    lods = lod_service.generate_lods(
        Path(data['file_path']),
        Path(data['file_path']).parent / 'lods'
    )
    return {**data, 'lods': lods}

@shared_task
def upload_to_cdn(data: dict):
    """上传到 CDN"""
    cdn = CDNService()
    urls = cdn.upload_batch(data['outputs'])
    return {**data, 'cdn_urls': urls}
```

## 6. 前端加载策略

### 6.1 渐进式加载

```typescript
// src/utils/model-loader.ts

class ProgressiveModelLoader {
  async load(url: string, options: LoadOptions) {
    // 1. 优先加载 LOD2（快速预览）
    const previewUrl = url.replace('.glb', '_lod2.glb');
    const previewModel = await this.loadModel(previewUrl);

    // 2. 后台加载 LOD1
    const standardUrl = url.replace('.glb', '_lod1.glb');
    this.loadModel(standardUrl).then(standardModel => {
      this.replaceModel(previewModel, standardModel);
    });

    // 3. 用户交互后加载 LOD0
    if (options.loadFullQuality) {
      const fullUrl = url.replace('.glb', '_lod0.glb');
      this.loadModel(fullUrl).then(fullModel => {
        this.replaceModel(previewModel, fullModel);
      });
    }
  }
}
```

### 6.2 预加载策略

```typescript
// src/utils/preload-manager.ts

class PreloadManager {
  private preloadQueue: string[] = [];

  // 基于视口距离的预加载
  observe(models: ModelInfo[]) {
    const observer = new IntersectionObserver((entries) => {
      entries.forEach(entry => {
        if (entry.isIntersecting) {
          const modelId = entry.target.getAttribute('data-model-id');
          this.preload(modelId);
        }
      });
    }, { rootMargin: '200px' });

    models.forEach(m => observer.observe(m.element));
  }

  async preload(modelId: string) {
    // 只加载 metadata 和 LOD2
    const meta = await fetch(`/api/v1/models/${modelId}/metadata`).then(r => r.json());
    const lod2Url = meta.lods[2];
    if (lod2Url) {
      await this.loadModel(lod2Url);
    }
  }
}
```

### 6.3 自适应加载

```typescript
// src/utils/adaptive-loader.ts

function getAdaptiveLOD(): number {
  const connection = (navigator as any).connection;

  // 根据网络状况选择 LOD
  if (connection) {
    if (connection.effectiveType === '4g' && !connection.saveData) {
      return 0; // 高质量
    }
    if (connection.effectiveType === '3g') {
      return 1; // 标准
    }
    return 2; // 低质量
  }

  // 根据设备性能
  const memory = (navigator as any).deviceMemory;
  if (memory && memory >= 8) return 0;
  if (memory && memory >= 4) return 1;
  return 2;
}
```
