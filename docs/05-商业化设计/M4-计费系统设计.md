### 3.1 核心表定义

```sql
-- plans 套餐表
CREATE TABLE plans (
    id          SERIAL PRIMARY KEY,
    name        VARCHAR(64) NOT NULL,
    description TEXT,
    price       DECIMAL(10, 2) NOT NULL,
    period      VARCHAR(20) DEFAULT 'month', -- month, year
    features    JSONB DEFAULT '[]',           -- 功能白名单
    quotas      JSONB DEFAULT '{}',           -- 配额配置
    sort_order  INT DEFAULT 0,
    is_active   BOOLEAN DEFAULT true,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- subscriptions 订阅表
CREATE TABLE subscriptions (
    id          SERIAL PRIMARY KEY,
    tenant_id   INT NOT NULL REFERENCES tenants(id),
    plan_id     INT NOT NULL REFERENCES plans(id),
    status      VARCHAR(20) DEFAULT 'active', -- active, expired, cancelled, suspended
    start_at    TIMESTAMP NOT NULL,
    end_at      TIMESTAMP NOT NULL,
    cancel_at   TIMESTAMP NULL,
    auto_renew  BOOLEAN DEFAULT true,
    created_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at  TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- usage_events 用量事件表 (写入 ClickHouse)
CREATE TABLE usage_events (
    id          UUID DEFAULT gen_random_uuid(),
    tenant_id   INT NOT NULL,
    metric      VARCHAR(64) NOT NULL, -- model_upload, spz_render, api_call
    value       BIGINT DEFAULT 1,
    timestamp   DateTime DEFAULT now(),
    metadata    JSONB
) ENGINE = MergeTree()
ORDER BY (tenant_id, metric, timestamp);
```

## 5. 用量计量方案

### 5.1 实时统计

```python
# backend/app/services/metering_service.py

import redis
from clickhouse_driver import Client

class MeteringService:
    def __init__(self):
        self.redis = redis.Redis()
        self.ch = Client(host='clickhouse')

    async def record_event(self, tenant_id: int, metric: str, value: int = 1, metadata: dict = None):
        # 1. Redis 实时计数（滑动窗口）
        key = f"usage:{tenant_id}:{metric}:{self.current_period()}"
        self.redis.incrby(key, value)
        self.redis.expire(key, 3600 * 24 * 35)  # 35天过期

        # 2. ClickHouse 持久化
        self.ch.execute(
            "INSERT INTO usage_events (tenant_id, metric, value, timestamp, metadata) VALUES",
            [(tenant_id, metric, value, datetime.utcnow(), json.dumps(metadata or {}))]
        )

    async def get_usage(self, tenant_id: int, metric: str, period: str = 'month') -> int:
        key = f"usage:{tenant_id}:{metric}:{self.period_key(period)}"
        val = self.redis.get(key)
        return int(val) if val else 0

    async def check_quota(self, tenant_id: int, metric: str) -> bool:
        quota = await self.get_quota(tenant_id, metric)
        usage = await self.get_usage(tenant_id, metric)
        return usage < quota
```

### 5.2 月度聚合（定时任务）

```python
# backend/app/scripts/aggregate_usage.py

async def aggregate_monthly():
    """每月1日00:00执行"""
    ch.execute("""
        INSERT INTO usage_aggregates
        SELECT
            tenant_id,
            metric,
            toStartOfMonth(timestamp) as period,
            sum(value) as total,
            now() as aggregated_at
        FROM usage_events
        WHERE timestamp >= toStartOfMonth(addMonths(now(), -1))
          AND timestamp < toStartOfMonth(now())
        GROUP BY tenant_id, metric, period
    """)
```

## 6. 配额网关设计

```python
# backend/app/core/quota_gateway.py

from fastapi import Request, HTTPException

class QuotaGateway:
    def __init__(self, metering: MeteringService):
        self.metering = metering

    async def check(self, request: Request, metric: str, cost: int = 1):
        tenant = request.state.tenant

        # 1. 查询配额
        quota = await self.metering.get_quota(tenant.id, metric)
        if quota is None:
            return  # 无限制

        # 2. 查询用量
        usage = await self.metering.get_usage(tenant.id, metric)

        # 3. 拦截超额
        if usage + cost > quota:
            raise HTTPException(
                status_code=403,
                detail={"code": 2002, "message": f"Quota exceeded for {metric}"}
            )

        # 4. 阈值告警（80%）
        if quota > 0 and usage / quota > 0.8:
            await self.send_alert(tenant.id, metric, usage, quota)

    async def consume(self, tenant_id: int, metric: str, cost: int = 1):
        await self.metering.record_event(tenant_id, metric, cost)
```

## 7. 退款/升降级规则

### 7.1 退款规则

| 场景 | 退款比例 | 说明 |
|------|----------|------|
| 7 天内未使用 | 100% | 全额退款 |
| 7 天内已使用 < 20% | 80% | 扣除手续费 |
| 7-30 天 | 按比例 | 未使用天数 / 总天数 |
| > 30 天 | 0% | 不支持退款 |

### 7.2 升降级规则

```
升级（低 → 高）：
  1. 计算当前套餐剩余价值
  2. 新套餐按比例折算
  3. 即时生效，新配额立即可用
  4. 账单周期不变

降级（高 → 低）：
  1. 当前周期内保持原套餐
  2. 下一周期开始时生效
  3. 超出新配额的资源标记为只读
```
