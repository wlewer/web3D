## 3. 后端优化清单

### 3.1 Redis 缓存

```python
# backend/app/core/cache.py

import json
import redis.asyncio as redis
from functools import wraps

redis_client = redis.Redis(host='localhost', port=6379, db=0)

def cached(expire: int = 300):
    def decorator(func):
        @wraps(func)
        async def wrapper(*args, **kwargs):
            cache_key = f"{func.__name__}:{hash(str(args))}:{hash(str(kwargs))}"
            cached = await redis_client.get(cache_key)
            if cached:
                return json.loads(cached)
            result = await func(*args, **kwargs)
            await redis_client.setex(cache_key, expire, json.dumps(result, default=str))
            return result
        return wrapper
    return decorator

# 使用示例
# backend/app/services/model_service.py

class ModelService:
    @cached(expire=600)
    async def get_model_list(self, tenant_id: int, page: int = 1, size: int = 20):
        # 查询数据库
        return await self.db.query(Model).filter_by(tenant_id=tenant_id).offset((page-1)*size).limit(size).all()
```

### 3.4 API 响应优化

```python
# backend/app/api/v1/endpoints/models.py

from fastapi import Query
from sqlalchemy import select
from sqlalchemy.orm import selectinload

@router.get("/models")
async def list_models(
    db: AsyncSession = Depends(get_db),
    tenant: Tenant = Depends(get_current_tenant),
    page: int = Query(1, ge=1),
    size: int = Query(20, ge=1, le=100),
    category: str = Query(None),
):
    query = (
        select(Model)
        .where(Model.tenant_id == tenant.id)
        .options(selectinload(Model.category))
        .order_by(Model.created_at.desc())
        .offset((page - 1) * size)
        .limit(size)
    )

    if category:
        query = query.where(Model.category_id == category)

    result = await db.execute(query)
    models = result.scalars().all()

    # 序列化时排除大字段
    return {
        "code": 0,
        "data": [
            {
                "id": m.id,
                "name": m.name,
                "thumbnail_url": m.thumbnail_url,
                "category": m.category.name if m.category else None,
                "created_at": m.created_at.isoformat(),
            }
            for m in models
        ],
    }
```

```typescript
// src/utils/three-optimizations.ts

export function optimizeRenderer(renderer: THREE.WebGLRenderer) {
  // 1. 启用深度缓冲优化
  renderer.sortObjects = false;

  // 2. 限制像素比
  const pixelRatio = Math.min(window.devicePixelRatio, 2);
  renderer.setPixelRatio(pixelRatio);

  // 3. 使用正确的颜色空间
  renderer.outputColorSpace = THREE.SRGBColorSpace;
}

export function optimizeScene(scene: THREE.Scene) {
  // 1. 静态对象合并
  const staticObjects: THREE.Mesh[] = [];
  scene.traverse((obj) => {
    if (obj.userData.isStatic && obj instanceof THREE.Mesh) {
      staticObjects.push(obj);
    }
  });

  if (staticObjects.length > 1) {
    const merged = mergeGeometries(staticObjects.map(m => m.geometry));
    const mesh = new THREE.Mesh(merged, staticObjects[0].material);
    scene.add(mesh);
    staticObjects.forEach(o => scene.remove(o));
  }

  // 2. 纹理复用
  const textureCache = new Map<string, THREE.Texture>();

  // 3. 视锥剔除
  scene.traverse((obj) => {
    if (obj instanceof THREE.Mesh) {
      obj.frustumCulled = true;
    }
  });
}

export function optimizeControls(controls: OrbitControls) {
  // 限制更新频率
  controls.enableDamping = true;
  controls.dampingFactor = 0.05;

  // 缩放限制
  controls.minDistance = 1;
  controls.maxDistance = 50;
}
```

### 4.3 SPZ 加载优化

```typescript
// src/utils/spz-loader.ts

class SPZLoader {
  private worker: Worker;

  constructor() {
    this.worker = new Worker(new URL('./spz-worker.ts', import.meta.url));
  }

  async load(url: string, onProgress?: (p: number) => void): Promise<THREE.Points> {
    // 1. 使用 Range 请求分块加载
    const response = await fetch(url, {
      headers: { 'Range': 'bytes=0-1048575' }, // 先加载前 1MB
    });

    // 2. Worker 中解析 SPZ
    return new Promise((resolve) => {
      this.worker.postMessage({ url, firstChunk: await response.arrayBuffer() });
      this.worker.onmessage = (e) => {
        if (e.data.type === 'progress') {
          onProgress?.(e.data.value);
        }
        if (e.data.type === 'complete') {
          resolve(this.createPoints(e.data.data));
        }
      };
    });
  }
}
```

## 5. Web Vitals 监控方案

```typescript
// src/utils/vitals-monitor.ts

import { getCLS, getFID, getFCP, getLCP, getTTFB, type Metric } from 'web-vitals';

class VitalsMonitor {
  private buffer: Metric[] = [];

  init() {
    getCLS((metric) => this.report(metric));
    getFID((metric) => this.report(metric));
    getFCP((metric) => this.report(metric));
    getLCP((metric) => this.report(metric));
    getTTFB((metric) => this.report(metric));

    // 自定义 3D 指标
    this.monitor3DPerf();
  }

  private report(metric: Metric) {
    this.buffer.push(metric);

    // 超过阈值立即上报
    if (metric.rating === 'poor') {
      this.sendImmediately(metric);
    }

    // 批量上报
    if (this.buffer.length >= 10) {
      this.flush();
    }
  }

  private monitor3DPerf() {
    let frameCount = 0;
    let lastTime = performance.now();

    const measure = () => {
      const now = performance.now();
      frameCount++;

      if (now - lastTime >= 1000) {
        const fps = frameCount;
        frameCount = 0;
        lastTime = now;

        if (fps < 24) {
          this.sendImmediately({
            name: '3d-fps',
            value: fps,
            rating: 'poor',
          } as any);
        }
      }

      requestAnimationFrame(measure);
    };

    requestAnimationFrame(measure);
  }

  private async sendImmediately(metric: Metric) {
    await fetch('/api/v1/metrics/vitals', {
      method: 'POST',
      body: JSON.stringify(metric),
      keepalive: true,
    });
  }

  private flush() {
    const data = [...this.buffer];
    this.buffer = [];

    navigator.sendBeacon('/api/v1/metrics/vitals/batch', JSON.stringify(data));
  }
}

export const vitalsMonitor = new VitalsMonitor();
```
