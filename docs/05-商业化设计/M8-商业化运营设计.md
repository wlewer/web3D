// src/utils/seo.ts

interface SEOConfig {
  title: string;
  description: string;
  keywords: string[];
  ogImage: string;
  canonical: string;
  structuredData?: object;
}

export function generateSEOMeta(config: SEOConfig) {
  return {
    title: `${config.title} | Web3D - 3D可视化平台`,
    meta: [
      { name: 'description', content: config.description },
      { name: 'keywords', content: config.keywords.join(', ') },
      { property: 'og:title', content: config.title },
      { property: 'og:description', content: config.description },
      { property: 'og:image', content: config.ogImage },
      { property: 'og:type', content: 'website' },
      { name: 'twitter:card', content: 'summary_large_image' },
      { rel: 'canonical', href: config.canonical },
    ],
    script: config.structuredData
      ? [{
          type: 'application/ld+json',
          innerHTML: JSON.stringify(config.structuredData),
        }]
      : [],
  };
}

### 5.2 实验配置 Schema

```typescript
interface Experiment {
  id: string;
  name: string;
  status: 'draft' | 'running' | 'paused' | 'completed';
  hypothesis: string;
  startDate: string;
  endDate?: string;
  trafficAllocation: number; // 参与实验的流量比例 (0-1)
  variants: Variant[];
  metrics: MetricConfig[];
  targeting?: TargetingRule[];
}

interface Variant {
  id: string;
  name: string;
  weight: number; // 流量权重
  config: Record<string, unknown>; // 功能开关配置
}

interface MetricConfig {
  name: string;
  type: 'conversion' | 'revenue' | 'engagement';
  event: string; // 追踪的事件名
  aggregation: 'sum' | 'average' | 'count' | 'unique';
}

interface TargetingRule {
  attribute: string; // 'tenant_plan', 'user_country', etc.
  operator: 'equals' | 'in' | 'gt' | 'lt';
  value: unknown;
}
```

### 5.3 流量分配算法

```typescript
// src/utils/experiment.ts

class ExperimentManager {
  private redis: Redis;

  async getVariant(experimentId: string, userId: string): Promise<string> {
    // 1. 检查用户是否已分配
    const cached = await this.redis.get(`exp:${experimentId}:${userId}`);
    if (cached) return cached;

    // 2. 获取实验配置
    const exp = await this.getExperiment(experimentId);
    if (!exp || exp.status !== 'running') return 'control';

    // 3. 一致性哈希分配
    const hash = this.hash(`${experimentId}:${userId}`);
    const bucket = hash % 100;

    let cumulative = 0;
    for (const variant of exp.variants) {
      cumulative += variant.weight * 100;
      if (bucket < cumulative) {
        await this.redis.setex(
          `exp:${experimentId}:${userId}`,
          86400 * 30, // 30天
          variant.id
        );
        return variant.id;
      }
    }

    return 'control';
  }

  private hash(str: string): number {
    let h = 0;
    for (let i = 0; i < str.length; i++) {
      h = ((h << 5) - h + str.charCodeAt(i)) | 0;
    }
    return Math.abs(h);
  }
}
```
