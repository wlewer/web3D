### 4.2 postMessage 协议

```typescript
// 消息格式
interface PluginMessage {
  id: string;
  type: 'request' | 'response' | 'event';
  namespace: 'storage' | 'canvas' | 'fetch' | 'model' | 'ui' | 'events' | 'log';
  action: string;
  payload?: unknown;
  error?: { code: string; message: string };
}

// 示例：存储读取
// iframe → parent
{ id: "msg_001", type: "request", namespace: "storage", action: "get", payload: { key: "settings" } }

// parent → iframe
{ id: "msg_001", type: "response", namespace: "storage", action: "get", payload: { value: { theme: "dark" } } }
```

### 4.3 安全策略

```typescript
// src/features/plugin-host/PluginSandbox.tsx

const PluginSandbox: React.FC<{ plugin: Plugin }> = ({ plugin }) => {
  const iframeRef = useRef<HTMLIFrameElement>(null);

  useEffect(() => {
    const handler = (e: MessageEvent) => {
      // 1. 校验 origin
      if (e.origin !== 'null' && !plugin.allowedOrigins.includes(e.origin)) {
        return;
      }

      // 2. 校验消息格式
      const msg = e.data as PluginMessage;
      if (!msg.id || !msg.namespace || !msg.action) {
        return;
      }

      // 3. 权限校验
      if (!plugin.permissions.includes(`${msg.namespace}:${msg.action}`)) {
        sendResponse(msg.id, { error: { code: 'PERMISSION_DENIED' } });
        return;
      }

      // 4. 路由到对应处理器
      handleMessage(msg).then(result => {
        sendResponse(msg.id, { payload: result });
      }).catch(err => {
        sendResponse(msg.id, { error: { code: 'INTERNAL_ERROR', message: err.message } });
      });
    };

    window.addEventListener('message', handler);
    return () => window.removeEventListener('message', handler);
  }, [plugin]);

  return (
    <iframe
      ref={iframeRef}
      src={plugin.entryUrl}
      sandbox="allow-scripts allow-same-origin"
      style={{ width: '100%', height: '100%', border: 'none' }}
    />
  );
};
```

## 5. 权限声明体系

| 权限 | 说明 | 风险等级 |
|------|------|----------|
| `storage:read` | 读取插件隔离存储 | 低 |
| `storage:write` | 写入插件隔离存储 | 低 |
| `canvas:read` | 读取画布 Schema | 低 |
| `canvas:write` | 修改画布节点 | 中 |
| `api:fetch` | 发起网络请求 | 中 |
| `tenant:read` | 读取租户信息 | 中 |
| `model:load` | 加载外部模型 | 中 |
| `user:read` | 读取用户信息 | 高 |
| `billing:read` | 读取计费信息 | 高 |

**权限申请流程：**

```
插件 manifest 声明权限
        │
        ▼
用户安装时弹窗展示权限列表
        │
        ▼
用户确认授权
        │
        ▼
运行时 SDK 拦截越权调用
        │
        ▼
越权调用 → 返回 PERMISSION_DENIED
```

```python
# backend/app/models/plugin.py

class Plugin(Base):
    __tablename__ = "plugins"

    id = Column(Integer, primary_key=True)
    manifest_id = Column(String(128), unique=True, nullable=False)
    name = Column(String(128), nullable=False)
    description = Column(Text)
    version = Column(String(32), nullable=False)
    author_id = Column(Integer, ForeignKey("users.id"))
    status = Column(String(20), default="pending")  # pending, approved, rejected, archived
    pricing_type = Column(String(20), default="free")  # free, paid, subscription
    price = Column(DECIMAL(10, 2), default=0)
    manifest_json = Column(JSON, nullable=False)
    package_url = Column(String(512))
    permissions = Column(JSON, default=list)
    category = Column(String(64))
    tags = Column(JSON, default=list)
    download_count = Column(Integer, default=0)
    rating = Column(DECIMAL(2, 1), default=5.0)
    created_at = Column(DateTime, default=datetime.utcnow)
    updated_at = Column(DateTime, default=datetime.utcnow, onupdate=datetime.utcnow)

class TenantPlugin(Base):
    __tablename__ = "tenant_plugins"

    id = Column(Integer, primary_key=True)
    tenant_id = Column(Integer, ForeignKey("tenants.id"))
    plugin_id = Column(Integer, ForeignKey("plugins.id"))
    installed_version = Column(String(32))
    config = Column(JSON, default=dict)
    is_enabled = Column(Boolean, default=True)
    installed_at = Column(DateTime, default=datetime.utcnow)
```

## 8. 安全扫描方案

| 扫描项 | 工具/方法 | 触发时机 |
|--------|----------|----------|
| JS 恶意代码检测 | ESLint Security + Semgrep | 提交时 |
| 依赖漏洞扫描 | Snyk / npm audit | 提交时 |
| XSS 载荷检测 | DOMPurify + 正则过滤 | 运行时 |
| 沙箱逃逸检测 | CSP + iframe sandbox | 运行时 |
| 网络请求审计 | 白名单校验 | 运行时 |
| WASM 代码扫描 | wasm-validate | 提交时 |

```yaml
# .github/workflows/plugin-security-scan.yml
name: Plugin Security Scan
on:
  pull_request:
    paths: ['plugins/**']
jobs:
  scan:
    runs-on: ubuntu-latest
    steps:
      - uses: actions/checkout@v4
      - name: Run Semgrep
        uses: returntocorp/semgrep-action@v1
        with:
          config: >-
            p/security-audit
            p/javascript
      - name: Audit Dependencies
        run: npm audit --audit-level=moderate
```
