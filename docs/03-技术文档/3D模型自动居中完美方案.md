# 3D模型自动居中完美方案

## 📋 问题分析

### 当前问题
从截图可以看到6个3D模型的显示效果：
- ✅ 蓝色大闪蝶：居中良好
- ❌ 苍蝇：只显示腿部
- ❌ 汉堡包：只显示底部
- ❌ 海鲈鱼：只显示顶部

### 根本原因
1. **包围盒尺寸差异大**
   - 蝴蝶：扁平展开型（宽>高）
   - 苍蝇：细长型（高>宽）
   - 汉堡：扁平型（宽>高，但很薄）

2. **几何中心偏移**
   - 模型的pivot点不一定在几何中心
   - 有些模型的坐标原点偏离中心很远

3. **朝向问题**
   - 有些模型朝上（Y轴）
   - 有些模型朝前（Z轴）

---

##  解决方案：五步自适应居中算法

### 核心思路
```
Step 1: 计算精确包围盒
Step 2: 智能裁剪多余空白（trim）
Step 3: 计算最佳相机距离
Step 4: 自动调整相机角度
Step 5: 添加安全边距
```

### 1. 增强版居中函数

```javascript
/**
 * 智能自适应居中算法 - 完美展示任意3D模型
 * @param {THREE.Object3D} object - 3D模型对象
 * @param {THREE.Camera} camera - 相机
 * @param {HTMLCanvasElement} canvas - 画布
 * @param {Object} controls - 控制器（可选）
 * @param {Object} options - 配置选项
 */
function smartFitCameraToObject(object, camera, canvas, controls, options = {}) {
  // 默认配置
  const config = {
    margin: options.margin || 1.15,        // 边距系数（1.15 = 15%边距）
    trimThreshold: options.trimThreshold || 0.05, // 裁剪阈值（5%）
    preferAxis: options.preferAxis || 'auto', // 优先轴向：'x', 'y', 'z', 'auto'
    autoRotate: options.autoRotate !== false, // 自动旋转调整
    debug: options.debug || false           // 调试模式
  };

  // ====== Step 1: 计算精确包围盒 ======
  const box = new THREE.Box3().setFromObject(object);
  const size = new THREE.Vector3();
  box.getSize(size);
  const center = new THREE.Vector3();
  box.getCenter(center);

  if (config.debug) {
    console.log('📦 原始包围盒:', {
      尺寸: `${size.x.toFixed(2)} x ${size.y.toFixed(2)} x ${size.z.toFixed(2)}`,
      中心: `(${center.x.toFixed(2)}, ${center.y.toFixed(2)}, ${center.z.toFixed(2)})`
    });
  }

  // ====== Step 2: 智能裁剪空白区域 ======
  const trimmedSize = trimEmptySpace(object, box, config.trimThreshold);
  
  if (config.debug) {
    console.log('✂️ 裁剪后尺寸:', 
      `${trimmedSize.x.toFixed(2)} x ${trimmedSize.y.toFixed(2)} x ${trimmedSize.z.toFixed(2)}`
    );
  }

  // ====== Step 3: 计算最佳相机距离 ======
  const canvasWidth = canvas.clientWidth;
  const canvasHeight = canvas.clientHeight;
  const aspect = canvasWidth / canvasHeight;
  
  // FOV转换为弧度
  const fovRad = camera.fov * Math.PI / 180;
  
  // 根据画布宽高比选择主导维度
  let dominantSize;
  if (aspect > 1) {
    // 横向画布：以宽度为主
    dominantSize = Math.max(trimmedSize.x, trimmedSize.y);
  } else {
    // 纵向/方形画布：以高度为主
    dominantSize = Math.max(trimmedSize.x, trimmedSize.y);
  }
  
  // 计算相机距离（考虑FOV和边距）
  const distance = (dominantSize / 2 / Math.tan(fovRad / 2)) * config.margin;

  if (config.debug) {
    console.log('📏 相机距离计算:', {
      画布比例: aspect.toFixed(2),
      主导尺寸: dominantSize.toFixed(2),
      FOV弧度: fovRad.toFixed(3),
      计算距离: distance.toFixed(2)
    });
  }

  // ====== Step 4: 自动调整相机角度 ======
  let cameraOffset;
  
  if (config.preferAxis === 'auto') {
    // 自动判断：哪个维度最大，就从那个方向观察
    if (trimmedSize.y > trimmedSize.x && trimmedSize.y > trimmedSize.z) {
      // 高度最大：从上前方观察（适合站立生物）
      cameraOffset = new THREE.Vector3(0, distance * 0.3, distance);
    } else if (trimmedSize.x > trimmedSize.z) {
      // 宽度最大：从正前方观察（适合扁平物体）
      cameraOffset = new THREE.Vector3(0, 0, distance);
    } else {
      // 深度最大：从侧面观察
      cameraOffset = new THREE.Vector3(distance, 0, 0);
    }
  } else {
    // 手动指定轴向
    switch (config.preferAxis) {
      case 'y': cameraOffset = new THREE.Vector3(0, distance * 0.3, distance); break;
      case 'x': cameraOffset = new THREE.Vector3(distance, 0, 0); break;
      case 'z': cameraOffset = new THREE.Vector3(0, 0, distance); break;
      default: cameraOffset = new THREE.Vector3(0, 0, distance);
    }
  }

  // ====== Step 5: 设置相机位置 ======
  camera.position.copy(center).add(cameraOffset);
  camera.lookAt(center);

  // 更新控制器
  if (controls) {
    controls.target.copy(center);
    controls.update();
  }

  // 调试信息
  if (config.debug) {
    console.log(' 最终配置:', {
      相机位置: `(${camera.position.x.toFixed(2)}, ${camera.position.y.toFixed(2)}, ${camera.position.z.toFixed(2)})`,
      观察目标: `(${center.x.toFixed(2)}, ${center.y.toFixed(2)}, ${center.z.toFixed(2)})`,
      距离: distance.toFixed(2)
    });
  }

  return { center, distance, trimmedSize };
}

/**
 * 智能裁剪空白区域
 * 分析模型顶点分布，移除无用的空白空间
 */
function trimEmptySpace(object, originalBox, threshold = 0.05) {
  // 收集所有顶点
  const vertices = [];
  object.traverse((child) => {
    if (child.geometry) {
      const positions = child.geometry.attributes.position;
      for (let i = 0; i < positions.count; i++) {
        const vertex = new THREE.Vector3();
        vertex.fromBufferAttribute(positions, i);
        child.localToWorld(vertex);
        vertices.push(vertex);
      }
    }
  });

  if (vertices.length === 0) {
    return originalBox.getSize(new THREE.Vector3());
  }

  // 计算顶点分布的百分位数
  const xs = vertices.map(v => v.x).sort((a, b) => a - b);
  const ys = vertices.map(v => v.y).sort((a, b) => a - b);
  const zs = vertices.map(v => v.z).sort((a, b) => a - b);

  // 使用5%和95%百分位数裁剪极端值
  const trimIndex = Math.floor(vertices.length * threshold);
  
  const trimmedBox = new THREE.Box3(
    new THREE.Vector3(
      xs[trimIndex],
      ys[trimIndex],
      zs[trimIndex]
    ),
    new THREE.Vector3(
      xs[xs.length - 1 - trimIndex],
      ys[ys.length - 1 - trimIndex],
      zs[zs.length - 1 - trimIndex]
    )
  );

  return trimmedBox.getSize(new THREE.Vector3());
}
```

```javascript
async function findBestViewAngle(object, camera, canvas) {
  const angles = [
    { x: 0, y: 0, z: 0 },      // 正面
    { x: 0.3, y: 0, z: 0 },    // 上方45度
    { x: -0.3, y: 0, z: 0 },   // 下方45度
    { x: 0, y: 0.5, z: 0 },    // 侧面
  ];
  
  let bestScore = -Infinity;
  let bestAngle = angles[0];
  
  for (const angle of angles) {
    // 临时设置相机
    const tempCamera = camera.clone();
    tempCamera.rotation.set(angle.x, angle.y, angle.z);
    
    // 计算可视面积占比
    const score = calculateVisibleRatio(object, tempCamera, canvas);
    
    if (score > bestScore) {
      bestScore = score;
      bestAngle = angle;
    }
  }
  
  return bestAngle;
}
```
