### 2.2 核心表结构

#### 2.2.1 组织表 (organizations)

```sql
CREATE TABLE organizations (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    name VARCHAR(255) NOT NULL,
    slug VARCHAR(100) UNIQUE NOT NULL,
    plan VARCHAR(50) DEFAULT 'free',  -- free, pro, enterprise
    quota_monthly INT DEFAULT 100,     -- 每月额度
    quota_used INT DEFAULT 0,
    settings JSONB DEFAULT '{}',
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    -- 约束
    CONSTRAINT chk_plan CHECK (plan IN ('free', 'pro', 'enterprise')),
    CONSTRAINT chk_quota CHECK (quota_used <= quota_monthly)
);

-- 索引
CREATE INDEX idx_org_slug ON organizations(slug);
CREATE INDEX idx_org_plan ON organizations(plan);
```

#### 2.2.2 用户表 (users)

```sql
CREATE TABLE users (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    org_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
    username VARCHAR(100) UNIQUE NOT NULL,
    email VARCHAR(255) UNIQUE NOT NULL,
    password_hash VARCHAR(255) NOT NULL,
    role VARCHAR(50) DEFAULT 'member',  -- admin, editor, member, viewer
    avatar_url VARCHAR(500),
    is_active BOOLEAN DEFAULT true,
    last_login_at TIMESTAMP,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    
    -- 约束
    CONSTRAINT chk_role CHECK (role IN ('admin', 'editor', 'member', 'viewer'))
);

-- 索引
CREATE INDEX idx_user_org ON users(org_id);
CREATE INDEX idx_user_email ON users(email);
CREATE INDEX idx_user_username ON users(username);
```

#### 2.2.3 3D引擎表 (engines) - 核心配置表

```sql
CREATE TABLE engines (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    code VARCHAR(50) UNIQUE NOT NULL,  -- hunyuan3d, trellis, gamba, triposr, mlsharp, vggt, sam3d
    name VARCHAR(100) NOT NULL,
    name_zh VARCHAR(100),              -- 中文名
    version VARCHAR(50) NOT NULL,
    type VARCHAR(50) NOT NULL,         -- generator, renderer, editor
    
    -- 能力配置
    capabilities JSONB NOT NULL DEFAULT '{
        "input_types": ["image"],
        "output_formats": ["glb", "ply", "obj"],
        "max_resolution": 2048,
        "supports_texture": true,
        "supports_pbr": false
    }',
    
    -- 资源配置
    resource_requirements JSONB NOT NULL DEFAULT '{
        "min_gpu_memory_gb": 8,
        "recommended_gpu": "NVIDIA A100",
        "min_inference_time_sec": 10,
        "max_queue_size": 100
    }',
    
    -- 模型配置
    model_config JSONB DEFAULT '{}',   -- HuggingFace路径、模型参数等
    
    -- 状态
    status VARCHAR(50) DEFAULT 'offline',  -- online, offline, maintenance, deprecated
    is_enabled BOOLEAN DEFAULT true,
    is_default BOOLEAN DEFAULT false,   -- 是否为默认引擎
    
    -- 排序
    display_order INT DEFAULT 0,
    
    -- 成本配置
    cost_per_task DECIMAL(10, 4) DEFAULT 0,  -- 每次任务成本(虚拟货币)
    daily_limit INT,                          -- 每日任务限制
    daily_used INT DEFAULT 0,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 索引
CREATE INDEX idx_engine_code ON engines(code);
CREATE INDEX idx_engine_status ON engines(status);
CREATE INDEX idx_engine_type ON engines(type);
CREATE INDEX idx_engine_enabled ON engines(is_enabled);
```

#### 2.2.4 GPU节点表 (gpu_nodes) - 动态管理GPU资源

```sql
CREATE TABLE gpu_nodes (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    engine_id UUID REFERENCES engines(id),
    
    -- 硬件信息
    hostname VARCHAR(100) NOT NULL,
    gpu_type VARCHAR(100) NOT NULL,          -- NVIDIA A100, RTX 3090, etc.
    gpu_count INT DEFAULT 1,
    gpu_memory_total_gb DECIMAL(10, 2) NOT NULL,
    gpu_memory_available_gb DECIMAL(10, 2),
    
    -- 负载信息
    current_utilization DECIMAL(5, 2) DEFAULT 0,  -- 0-100%
    current_memory_used_gb DECIMAL(10, 2) DEFAULT 0,
    active_tasks INT DEFAULT 0,
    max_concurrent_tasks INT DEFAULT 2,
    
    -- 状态
    status VARCHAR(50) DEFAULT 'offline',  -- online, offline, busy, error
    health_status VARCHAR(50) DEFAULT 'healthy',  -- healthy, warning, critical
    
    -- 健康检查
    last_heartbeat TIMESTAMP,
    error_count INT DEFAULT 0,
    last_error TEXT,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 索引
CREATE INDEX idx_gpu_engine ON gpu_nodes(engine_id);
CREATE INDEX idx_gpu_status ON gpu_nodes(status);
CREATE INDEX idx_gpu_available ON gpu_nodes(gpu_memory_available_gb);
```

#### 2.2.5 资产表 (assets) - 存储生成的3D资产

```sql
CREATE TABLE assets (
    id UUID PRIMARY KEY DEFAULT gen_random_uuid(),
    user_id UUID REFERENCES users(id) ON DELETE CASCADE,
    org_id UUID REFERENCES organizations(id) ON DELETE CASCADE,
    engine_id UUID REFERENCES engines(id),
    
    -- 基本信息
    name VARCHAR(255) NOT NULL,
    description TEXT,
    
    -- 文件信息
    input_file_url VARCHAR(500),      -- 输入图像URL
    input_file_size INT,               -- 输入文件大小(bytes)
    output_file_url VARCHAR(500),      -- 输出3D文件URL
    output_file_size INT,              -- 输出文件大小(bytes)
    output_format VARCHAR(20),         -- glb, ply, obj, spz
    
    -- 技术信息
    input_metadata JSONB DEFAULT '{}',   -- 输入图像信息
    output_metadata JSONB DEFAULT '{}',   -- 3D模型信息(顶点数、面数等)
    render_config JSONB DEFAULT '{}',    -- 渲染配置
    
    -- 质量评估
    quality_score DECIMAL(5, 2),          -- 0-100质量评分
    generation_time_ms INT,               -- 生成耗时
    
    -- 状态
    status VARCHAR(50) DEFAULT 'pending',  -- pending, processing, completed, failed, deleted
    is_public BOOLEAN DEFAULT false,
    is_featured BOOLEAN DEFAULT false,
    
    -- 统计
    view_count INT DEFAULT 0,
    download_count INT DEFAULT 0,
    
    -- 标签
    tags TEXT[] DEFAULT '{}',
    
    -- 时间戳
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    deleted_at TIMESTAMP
);

-- 索引
CREATE INDEX idx_asset_user ON assets(user_id);
CREATE INDEX idx_asset_org ON assets(org_id);
CREATE INDEX idx_asset_engine ON assets(engine_id);
CREATE INDEX idx_asset_status ON assets(status);
CREATE INDEX idx_asset_created ON assets(created_at DESC);
CREATE INDEX idx_asset_tags ON assets USING GIN(tags);
```

#### 2.2.6 任务表 (tasks) - 跟踪3D生成任务

```sql
CREATE TABLE tasks (

    engine_id UUID REFERENCES engines(id),
    
    -- 任务信息

    priority INT DEFAULT 5,            -- 1-10, 10最高
    
    -- 参数配置

    output_params JSONB DEFAULT '{}',
    
    -- 执行状态
    status VARCHAR(50) DEFAULT 'queued',  -- queued, running, completed, failed, cancelled

    progress_message VARCHAR(255),
    
    -- 时间戳
    queued_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    started_at TIMESTAMP,
    completed_at TIMESTAMP,
    
    -- 错误信息
    error_code VARCHAR(50),
    error_message TEXT,
    retry_count INT DEFAULT 0,
    max_retries INT DEFAULT 3,
    
    -- 资源消耗
    gpu_memory_used_gb DECIMAL(10, 2),
    execution_time_ms INT,
    
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 索引
CREATE INDEX idx_task_asset ON tasks(asset_id);
CREATE INDEX idx_task_user ON tasks(user_id);
CREATE INDEX idx_task_gpu ON tasks(gpu_node_id);
CREATE INDEX idx_task_status ON tasks(status);
CREATE INDEX idx_task_queued ON tasks(queued_at);
```

```sql
CREATE INDEX idx_log_created ON logs(created_at DESC);

-- 分区表(按月分区，提高查询性能)
CREATE TABLE logs_partitioned () INHERITS (logs);

-- 创建自动分区函数
CREATE OR REPLACE FUNCTION create_log_partition()
RETURNS void AS $$
DECLARE
    partition_date DATE;
    partition_name TEXT;
BEGIN
    partition_date := CURRENT_DATE;
    partition_name := 'logs_' || to_char(partition_date, 'YYYY_MM');
    
    EXECUTE format(
        'CREATE TABLE IF NOT EXISTS %I PARTITION OF logs_partitioned 
         FOR VALUES FROM (%L) TO (%L)',
        partition_name,
        partition_date,
        partition_date + INTERVAL '1 month'
    );
END;
$$ LANGUAGE plpgsql;
```

```typescript
// src/services/api/index.ts - API配置
export const API_CONFIG = {
  baseURL: import.meta.env.VITE_API_BASE_URL || '/api/v1',
  timeout: 30000,
  retryAttempts: 3,

// src/services/api/engines.ts - 引擎API服务
import { apiClient } from './client';

export interface Engine {
  id: string;
  code: string;
  name: string;
  name_zh: string;
  version: string;
  type: 'generator' | 'renderer' | 'editor';
  capabilities: {
    input_types: string[];
    output_formats: string[];
    max_resolution: number;
    supports_texture: boolean;
    supports_pbr: boolean;
  };
  resource_requirements: {
    min_gpu_memory_gb: number;
    recommended_gpu: string;
    min_inference_time_sec: number;
  };
  status: 'online' | 'offline' | 'maintenance';
  is_default: boolean;
  queue_size: number;
  estimated_time_sec: number;
}

export const enginesAPI = {
  // 获取可用引擎列表 - 动态从后端获取，不再静态缓存
  listEngines: async (params?: {
    type?: string;
    status?: string;
    page?: number;
    pageSize?: number;
  }): Promise<{ engines: Engine[]; total: number }> => {
    const response = await apiClient.get('/engines', { params });
    return response.data;
  },

  // 获取单个引擎详情
  getEngine: async (code: string): Promise<Engine> => {
    const response = await apiClient.get(`/engines/${code}`);
    return response.data;
  },

  // 获取引擎健康状态 - 实时查询
  getEngineHealth: async (code: string): Promise<{
    status: string;
    queue_size: number;
    avg_wait_time: number;
    gpu_nodes: Array<{
      id: string;
      status: string;
      utilization: number;
    }>;
  }> => {
    const response = await apiClient.get(`/engines/${code}/health`);
    return response.data;
  },

  // 获取引擎能力配置
  getEngineCapabilities: async (code: string): Promise<Engine['capabilities']> => {
    const response = await apiClient.get(`/engines/${code}/capabilities`);
    return response.data;
  },
};

// src/stores/engines.ts - 引擎状态管理(Zustand)
import { create } from 'zustand';
import { enginesAPI, type Engine } from '../services/api/engines';

interface EnginesState {
  engines: Engine[];
  selectedEngine: Engine | null;
  isLoading: boolean;
  error: string | null;
  lastFetchTime: number | null;
  
  // 缓存策略: 5分钟过期
  CACHE_DURATION: 5 * 60 * 1000;
  
  // Actions
  fetchEngines: () => Promise<void>;
  selectEngine: (code: string) => void;
  getEngineHealth: (code: string) => Promise<any>;
}

export const useEnginesStore = create<EnginesState>((set, get) => ({
  engines: [],
  selectedEngine: null,
  isLoading: false,
  error: null,
  lastFetchTime: null,
  
  fetchEngines: async () => {
    const { CACHE_DURATION, lastFetchTime } = get();
    
    // 检查缓存是否过期
    if (lastFetchTime && Date.now() - lastFetchTime < CACHE_DURATION) {
      return; // 缓存有效，不重新获取
    }
    
    set({ isLoading: true, error: null });
    
    try {
      const data = await enginesAPI.listEngines();
      set({ 
        engines: data.engines, 
        lastFetchTime: Date.now(),
        isLoading: false 
      });
    } catch (error: any) {
      set({ 
        error: error.message || '获取引擎列表失败',
        isLoading: false 
      });
    }
  },
  
  selectEngine: (code: string) => {
    const { engines } = get();
    const engine = engines.find(e => e.code === code);
    set({ selectedEngine: engine || null });
  },
  
  getEngineHealth: async (code: string) => {
    try {
      return await enginesAPI.getEngineHealth(code);
    } catch (error) {
      console.error('获取引擎健康状态失败:', error);
      return null;
    }
  },
}));
```

### 6.1 避免静态缓存的核心策略

| 问题 | 解决方案 |
|:---|:---|
| 引擎列表静态硬编码 | 后端数据库管理，前端动态获取 |

| 模型URL过期 | 签名URL+过期时间，后端刷新 |

```typescript
// 引擎选择时的数据一致性检查
async function selectBestEngine(): Promise<Engine> {
  // 1. 获取所有可用引擎
  const { engines } = await enginesAPI.listEngines();
  
  // 2. 过滤在线引擎
  const onlineEngines = engines.filter(e => e.status === 'online');
  
  // 3. 获取每个引擎的实时健康状态
  const engineHealthPromises = onlineEngines.map(async (e) => {
    const health = await enginesAPI.getEngineHealth(e.code);
    return { engine: e, health };
  });
  
  const engineHealthList = await Promise.all(engineHealthPromises);
  
  // 4. 选择最优引擎(队列最短 + 负载最低)
  const best = engineHealthList.sort((a, b) => {
    const aScore = a.health.queue_size * 10 + (100 - a.health.gpu_nodes[0]?.utilization || 100);
    const bScore = b.health.queue_size * 10 + (100 - b.health.gpu_nodes[0]?.utilization || 100);
    return aScore - bScore;
  })[0];
  
  return best.engine;
}
```
