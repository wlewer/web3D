## 📋 执行摘要

- ✅ 组件设计良好，Props接口清晰，支持多种布局模式

### 2. 主要问题识别

#### 问题1：装饰性功能耦合
```typescript
// 当前实现：装饰性功能直接写在组件内部
showParticles?: boolean;   // 粒子背景
showPlatform?: boolean;    // 展示台

```

**影响**：
- 在纯3D查看场景（如后台管理）中，这些功能是冗余的
- 增加了不必要的渲染开销
- 降低了组件的通用性

#### 问题2：布局模式硬编码
```typescript
// 当前实现：布局样式通过className硬编码
const layoutClasses = {
  featured: 'ugc-v2-layout-featured',
  grid: 'ugc-v2-layout-grid',
  list: 'ugc-v2-layout-list',
  // ...

```

**影响**：
- 难以自定义新的布局模式
- CSS样式与组件逻辑耦合
- 无法动态切换布局

#### 问题3：智能居中算法不可复用
```typescript
// 当前实现：smartFitCameraToObject函数内嵌在组件中
const smartFitCameraToObject = useCallback((object, camera, canvas, config) => {
  // 100+行的复杂逻辑
```

**影响**：
- 其他组件无法复用此算法
- 代码重复（如果其他组件需要类似功能）
- 难以独立测试和优化

#### 问题4：缺少轻量级版本
- 对于简单的3D预览场景（如缩略图、列表项），V2组件过于重量级
- 开发者可能选择继续使用旧版或自定义实现

#### 1.2 创建模型加载器

```typescript
import * as THREE from 'three';
import { SplatMesh } from '@sparkjsdev/spark';
import { GLTFLoader } from 'three/examples/jsm/loaders/GLTFLoader.js';
import { PLYLoader } from 'three/examples/jsm/loaders/PLYLoader.js';

export type ModelFormat = 'spz' | 'glb' | 'gltf' | 'ply';

export interface LoadProgress {
  progress: number;
  stage: 'initializing' | 'loading' | 'processing' | 'rendering';
}

export interface LoadResult {
  model: THREE.Object3D | SplatMesh;
  format: ModelFormat;
  pointCount?: number;
}

/**
 * 统一模型加载器 - 支持多种格式
 */
export class ModelLoader {
  private static loaders = new Map<ModelFormat, any>();

  /**
   * 加载模型
   */
  static async load(url: string, onProgress?: (progress: LoadProgress) => void): Promise<LoadResult> {
    const format = this.detectFormat(url);
    
    switch (format) {
      case 'spz':
        return this.loadSPZ(url, onProgress);
      case 'glb':
      case 'gltf':
        return this.loadGLB(url, onProgress);
      case 'ply':
        return this.loadPLY(url, onProgress);
      default:
        throw new Error(`Unsupported format: ${format}`);
    }
  }

  /**
   * 检测模型格式
   */
  private static detectFormat(url: string): ModelFormat {
    const ext = url.split('.').pop()?.toLowerCase();
    if (ext === 'spz') return 'spz';
    if (ext === 'glb') return 'glb';
    if (ext === 'gltf') return 'gltf';
    if (ext === 'ply') return 'ply';
    return 'spz'; // 默认
  }

  /**
   * 加载SPZ格式
   */
  private static async loadSPZ(url: string, onProgress?: (progress: LoadProgress) => void): Promise<LoadResult> {
    onProgress?.({ progress: 0, stage: 'initializing' });
    
    const response = await fetch(url);
    const arrayBuffer = await response.arrayBuffer();
    
    onProgress?.({ progress: 50, stage: 'loading' });
    
    const splat = new SplatMesh();
    await splat.load(arrayBuffer);
    
    onProgress?.({ progress: 100, stage: 'rendering' });
    
    return {
      model: splat,
      format: 'spz',
      pointCount: splat.splatCount
    };
  }

  /**
   * 加载GLB/GLTF格式
   */
  private static async loadGLB(url: string, onProgress?: (progress: LoadProgress) => void): Promise<LoadResult> {
    onProgress?.({ progress: 0, stage: 'initializing' });
    
    const loader = new GLTFLoader();
    
    return new Promise((resolve, reject) => {
      loader.load(
        url,
        (gltf) => {
          onProgress?.({ progress: 100, stage: 'rendering' });
          resolve({
            model: gltf.scene,
            format: url.endsWith('gltf') ? 'gltf' : 'glb'
          });
        },
        (xhr) => {
          const progress = (xhr.loaded / xhr.total) * 100;
          onProgress?.({ progress, stage: 'loading' });
        },
        reject
      );
    });
  }
```

```typescript
import * as THREE from 'three';
import { OrbitControls } from 'three/examples/jsm/controls/OrbitControls.js';

export interface CameraConfig {
  position: [number, number, number];
  target: [number, number, number];
  zoom: number;
}

export interface CameraOptions {
  fov?: number;
  near?: number;
  far?: number;
  enableControls?: boolean;
  autoRotate?: boolean;
}

/**
 * 相机管理器 - 统一管理相机创建和控制
 */
export class CameraManager {
  /**
   * 创建相机
   */
  static createCamera(
    canvas: HTMLCanvasElement,
    options: CameraOptions = {}
  ): { camera: THREE.PerspectiveCamera; controls: OrbitControls | null } {
    const aspect = canvas.clientWidth / canvas.clientHeight;
    const camera = new THREE.PerspectiveCamera(
      options.fov || 60,
      aspect,
      options.near || 0.1,
      options.far || 1000
    );

    let controls: OrbitControls | null = null;
    if (options.enableControls !== false) {
      controls = new OrbitControls(camera, canvas);
      controls.enableDamping = true;
      controls.dampingFactor = 0.05;
      controls.autoRotate = options.autoRotate || false;
    }

    return { camera, controls };
  }

  /**
   * 应用相机配置
   */
  static applyConfig(
    camera: THREE.PerspectiveCamera,
    controls: OrbitControls | null,
    config: CameraConfig
  ) {
    camera.position.set(...config.position);
    if (controls) {
      controls.target.set(...config.target);
      controls.update();
    }
  }

  /**
   * 保存当前相机配置
   */
  static saveConfig(camera: THREE.PerspectiveCamera, controls: OrbitControls | null): CameraConfig {
    return {
      position: [camera.position.x, camera.position.y, camera.position.z],
      target: controls ? [
        controls.target.x,
        controls.target.y,
        controls.target.z

      zoom: camera.zoom
```

```typescript
import React, { useEffect, useRef, useState, useCallback, forwardRef, useImperativeHandle } from 'react';
import * as THREE from 'three';
import { SmartCenteringEngine } from './engines/SmartCenteringEngine';
import { ModelLoader, LoadProgress, LoadResult } from './engines/ModelLoader';
import { CameraManager, CameraConfig } from './engines/CameraManager';

export interface Base3DViewerProps {
  modelUrl: string;
  autoCenter?: boolean;
  margin?: number;
  autoRotate?: boolean;
  enableControls?: boolean;
  backgroundColor?: string;
  onLoadComplete?: () => void;
  onError?: (error: Error) => void;
  onProgress?: (progress: number) => void;
}

export interface Base3DViewerRef {
  getModel: () => THREE.Object3D | null;
  reload: () => void;
  toggleAutoRotate: () => void;
  screenshot: () => string;
  saveCameraConfig: () => CameraConfig;
  loadCameraConfig: (config: CameraConfig) => void;
  resetCamera: () => void;
}

/**
 * 基础3D查看器 - 纯渲染功能，无装饰
 */
export const Base3DViewer = forwardRef<Base3DViewerRef, Base3DViewerProps>(({
  modelUrl,
  autoCenter = true,
  margin = 2.5,
  autoRotate = true,
  enableControls = true,
  backgroundColor = '#0a0a0f',
  onLoadComplete,
  onError,
  onProgress,
}, ref) => {
  const containerRef = useRef<HTMLDivElement>(null);
  const canvasRef = useRef<HTMLCanvasElement>(null);
  
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState<string | null>(null);

  // 初始化Three.js场景
  useEffect(() => {
    if (!containerRef.current || !canvasRef.current) return;

    const { camera, controls } = CameraManager.createCamera(canvasRef.current, {
      enableControls,
      autoRotate
    });

    const scene = new THREE.Scene();
    const renderer = new THREE.WebGLRenderer({ 
      canvas: canvasRef.current, 
      antialias: true,
      alpha: true 
    });

    // 清理函数
    return () => {
      renderer.dispose();
      controls?.dispose();
    };
  }, [enableControls, autoRotate]);

  // 加载模型
  useEffect(() => {
    const loadModel = async () => {
      try {
        setLoading(true);
        setError(null);

        const result = await ModelLoader.load(modelUrl, (progress) => {
          onProgress?.(progress.progress);
        });

        // 添加到场景
        // 智能居中
        if (autoCenter && canvasRef.current) {
          const fitResult = SmartCenteringEngine.calculateFit(
            result.model,
            canvasRef.current,
            { margin, trimThreshold: 0.05, preferAxis: 'auto', autoCenter: true }
          );
          // 应用相机位置
        }

        setLoading(false);
        onLoadComplete?.();
      } catch (err) {
        setError(err instanceof Error ? err.message : 'Unknown error');
        onError?.(err instanceof Error ? err : new Error(String(err)));
      }
    };

    loadModel();
  }, [modelUrl, autoCenter, margin]);

  // 暴露方法给父组件
  useImperativeHandle(ref, () => ({
    getModel: () => null, // 实际实现
    reload: () => {},
    toggleAutoRotate: () => {},
    screenshot: () => '',
    saveCameraConfig: () => ({ position: [0, 0, 0], target: [0, 0, 0], zoom: 1 }),
    loadCameraConfig: () => {},
    resetCamera: () => {}
  }));

  return (
    <div 
      ref={containerRef} 
      style={{ width: '100%', height: '100%', backgroundColor }}
    >
      <canvas ref={canvasRef} style={{ width: '100%', height: '100%' }} />
      {loading && <div className="loading-overlay">Loading...</div>}
      {error && <div className="error-overlay">{error}</div>}
    </div>
  );
});
```
